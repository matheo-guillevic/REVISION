\section*{Partie I : Conception d'un registre [40/90] \cite{1059}}

\subsection*{Question 1.1 \cite{1060}}
\textbf{Énoncé :} Dessinez en utilisant uniquement des bascules DFF (possédant des entrées reset et enable) le circuit d'un registre $n$ bits à chargement parallèle et sortie parallèle capturant de l'entrée \texttt{Din} sur la sortie \texttt{Dout} sur les fronts montants de l'horloge \texttt{clk} avec \cite{1060} :
\begin{itemize}
    \item une entrée \texttt{enable} active à l'état haut autorisant le chargement de l'entrée \texttt{Din} \cite{1061}.
    \item une entrée de données \texttt{Din} sur $n$ bits \cite{1062}.
    \item une sortie \texttt{Dout} sur $n$ bits \cite{1063}.
    \item une remise à zéro \texttt{Reset} synchrone active à l'état haut \cite{1064}.
\end{itemize}

\textbf{Correction et Explication :}
Un registre sur $n$ bits est simplement une mise en parallèle de $n$ bascules D. 
Pour concevoir ce circuit :
\begin{itemize}
    \item Tous les signaux de contrôle (\texttt{clk}, \texttt{reset}, \texttt{enable}) sont connectés en commun à toutes les bascules.
    \item Le bit $i$ de l'entrée de données (\texttt{Din[i]}) est connecté à l'entrée \texttt{D} de la bascule $i$.
    \item Le bit $i$ de la sortie de données (\texttt{Dout[i]}) est connecté à la sortie \texttt{Q} de la bascule $i$.
\end{itemize}
\textit{Note : Lors de la capture d'un front montant d'horloge, si \texttt{reset} est à '1', la sortie passe à 0. Sinon, si \texttt{enable} est à '1', l'entrée \texttt{D} est recopiée sur la sortie \texttt{Q}.}

\subsection*{Question 1.2 \cite{1065}}
\textbf{Énoncé :} Ecrire en VHDL RTL l'entité générique (\texttt{registerN}) et l'architecture (\texttt{rtl}) du composant précédent [10] \cite{1065}.

\textbf{Correction et Explication :}
La description "RTL" (Register Transfer Level) décrit le comportement du registre en utilisant un processus synchrone. Le \texttt{reset} étant synchrone, il doit être testé à l'intérieur de la condition du front d'horloge \cite{1064}.

\begin{lstlisting}
library IEEE;
use IEEE.std_logic_1164.all;

entity registerN is
    generic (
        n : integer := 8 -- Valeur par defaut arbitraire
    );
    port (
        clk    : in  std_logic;
        reset  : in  std_logic;
        enable : in  std_logic;
        Din    : in  std_logic_vector(n-1 downto 0);
        Dout   : out std_logic_vector(n-1 downto 0)
    );
end entity registerN;

architecture rtl of registerN is
begin
    process(clk)
    begin
        if rising_edge(clk) then
            if reset = '1' then
                Dout <= (others => '0'); -- Remise a zero synchrone
            elsif enable = '1' then
                Dout <= Din;             -- Chargement parallele
            end if;
        end if;
    end process;
end architecture rtl;
\end{lstlisting}

\subsection*{Question 1.3 \cite{1066}}
\textbf{Énoncé :} Combien de bascules seront générées par la synthèse de cette description rtl [5] \cite{1066}.

\textbf{Correction et Explication :}
La synthèse logique générera exactement \textbf{$n$ bascules D}. 
\textit{Explication :} L'entrée et la sortie étant définies par des vecteurs de taille $n$ (\texttt{n-1 downto 0}), le processus assigne une valeur mémorisée à chaque bit du vecteur \texttt{Dout} à chaque front d'horloge. Chaque bit mémorisé correspond physiquement à une bascule.

\subsection*{Question 1.4 \cite{1067}}
\textbf{Énoncé :} Ecrire en VHDL structurel l'architecture (\texttt{struct}) de la même entité (\texttt{registerN}) précédente [15] \cite{1067}. On dispose du composant suivant \cite{1068, 1070-1073} :
\begin{lstlisting}
entity bascule is
port(d, clk, reset, enable: in std_logic;
     q: out std_logic);
end entity bascule;
\end{lstlisting}

\textbf{Correction et Explication :}
Une description structurelle consiste à instancier et relier des sous-composants existants (ici, l'entité \texttt{bascule}). Puisque le registre a une taille générique $n$, nous devons utiliser une boucle \texttt{generate} pour instancier les $n$ bascules de façon élégante.

\begin{lstlisting}
architecture struct of registerN is
    -- Declaration du composant a instancier
    component bascule
        port(d, clk, reset, enable: in std_logic;
             q: out std_logic);
    end component;
begin
    -- Generation de n bascules
    gen_bascules: for i in 0 to n-1 generate
        bascule_inst : bascule
            port map (
                d      => Din(i),
                clk    => clk,
                reset  => reset,
                enable => enable,
                q      => Dout(i)
            );
    end generate gen_bascules;
end architecture struct;
\end{lstlisting}

\section*{Partie II : Conception d'un « compteur générique » [50/90] \cite{1074}}

\subsection*{Question 2.1 \cite{1075}}
\textbf{Énoncé :} Ecrire en VHDL RTL l'entité (\texttt{compteur\_moduloN}) avec un paramètre générique \texttt{N} et l'architecture (\texttt{rtl}) d'un compteur comptant sur 4 bits et modulo N [15] \cite{1075}.
\begin{itemize}
    \item Entrées : horloge (\texttt{clk}) sur front montant, reset (\texttt{rst}) asynchrone actif haut, enable (\texttt{en}) actif haut \cite{1077}.
    \item Sorties : valeur (\texttt{count}, sur 4 bits), signal (\texttt{fin}) qui passe à '1' quand la valeur max (N-1) est atteinte \cite{1078}.
\end{itemize}

\textbf{Correction et Explication :}
Le compteur est "modulo N", ce qui signifie qu'il compte de $0$ à $N-1$, puis repasse à $0$. Le reset asynchrone implique que le signal \texttt{rst} doit figurer dans la liste de sensibilité du processus et être testé avant le front d'horloge.

\begin{lstlisting}
library IEEE;
use IEEE.std_logic_1164.all;
use IEEE.numeric_std.all; -- Utilisation recommandee pour les compteurs

entity compteur_moduloN is
    generic (
        N : integer := 10 -- Modulo par defaut
    );
    port (
        clk   : in  std_logic;
        rst   : in  std_logic;
        en    : in  std_logic;
        count : out std_logic_vector(3 downto 0);
        fin   : out std_logic
    );
end entity compteur_moduloN;

architecture rtl of compteur_moduloN is
    -- Signal interne pour faciliter les operations arithmetiques
    signal count_int : unsigned(3 downto 0);
begin
    process(clk, rst)
    begin
        -- Reset asynchrone
        if rst = '1' then
            count_int <= (others => '0');
        -- Action synchrone sur front montant
        elsif rising_edge(clk) then
            if en = '1' then
                if count_int = N - 1 then
                    count_int <= (others => '0'); -- Bouclage modulo N
                else
                    count_int <= count_int + 1;   -- Incrementation
                end if;
            end if;
        end if;
    end process;

    -- Assignations concurrentes pour les sorties
    count <= std_logic_vector(count_int);
    
    -- Le signal fin vaut '1' uniquement quand le compteur atteint N-1
    fin <= '1' when (count_int = N - 1) else '0';
    
end architecture rtl;
\end{lstlisting}

\subsection*{Question 2.2 \cite{1079}}
\textbf{Énoncé :} Combien de bascules seront générées par la synthèse logique de votre description précédente [5] ? \cite{1079}

\textbf{Correction et Explication :}
\textbf{4 bascules} seront générées.
\textit{Explication :} L'énoncé précise explicitement que la sortie du compteur (\texttt{count}) est sur 4 bits \cite{1078}. Même si $N$ est générique, la taille du registre interne mémorisant l'état est fixée à 4 bits (de 0 à 15 maximum). Le signal \texttt{fin} est un signal purement combinatoire généré à partir de ces 4 bascules (il n'est pas mémorisé).

\subsection*{Question 2.3 \cite{1080}}
\textbf{Énoncé :} Ecrire en VHDL un testbench permettant de simuler l'architecture précédente pour un modulo $N=8$ [15] \cite{1080}.

\textbf{Correction et Explication :}
Un banc de test (testbench) n'a ni entrée ni sortie. Il instancie le composant à tester (UUT - Unit Under Test), génère l'horloge et applique des stimuli (scénario de test) sur les entrées pour observer le comportement.

\begin{lstlisting}
library IEEE;
use IEEE.std_logic_1164.all;

entity tb_compteur is
-- Un testbench n'a pas de ports
end entity tb_compteur;

architecture behavior of tb_compteur is
    -- Signaux pour se connecter a l'UUT
    signal clk   : std_logic := '0';
    signal rst   : std_logic := '0';
    signal en    : std_logic := '0';
    signal count : std_logic_vector(3 downto 0);
    signal fin   : std_logic;

    constant clk_period : time := 10 ns;

begin
    -- Instanciation de l'UUT avec N = 8
    uut: entity work.compteur_moduloN
        generic map (
            N => 8
        )
        port map (
            clk   => clk,
            rst   => rst,
            en    => en,
            count => count,
            fin   => fin
        );

    -- Generation de l'horloge
    clk_process : process
    begin
        clk <= '0';
        wait for clk_period/2;
        clk <= '1';
        wait for clk_period/2;
    end process;

    -- Processus des stimuli
    stim_proc: process
    begin
        -- Initialisation : Reset actif
        rst <= '1';
        en  <= '0';
        wait for 20 ns;
        
        -- Relachement du reset, activation du compteur
        rst <= '0';
        en  <= '1';
        
        -- On laisse tourner le compteur pour voir au moins un cycle complet (N=8)
        -- 8 coups d'horloge font 80 ns. On attend 100 ns.
        wait for 100 ns;
        
        -- Fin de la simulation
        en <= '0';
        wait;
    end process;
end architecture behavior;
\end{lstlisting}

\subsection*{Question 2.4 \cite{1081}}
\textbf{Énoncé :} Dessinez les chronogrammes des signaux de l'entité correspondant très exactement à la simulation que donnerait votre testbench précédent [15] \cite{1081}. Indiquez clairement l'échelle des temps sur l'axe temporel \cite{1082}.

\textbf{Correction et Explication :}
Voici la description textuelle du chronogramme généré par le banc de test précédent.

\begin{itemize}
    \item \textbf{De 0 ns à 20 ns :}
    \begin{itemize}
        \item \texttt{rst} = '1', \texttt{en} = '0'
        \item \texttt{count} = "0000" (0 en décimal, maintenu par le reset asynchrone)
        \item \texttt{fin} = '0'
    \end{itemize}
    \item \textbf{À 20 ns :} \texttt{rst} descend à '0', \texttt{en} passe à '1'.
    \item \textbf{De 20 ns à 100 ns (Activité de comptage) :}
    \begin{itemize}
        \item À 25 ns (front montant) : \texttt{count} $\rightarrow$ 1
        \item À 35 ns (front montant) : \texttt{count} $\rightarrow$ 2
        \item À 45 ns (front montant) : \texttt{count} $\rightarrow$ 3
        \item À 55 ns (front montant) : \texttt{count} $\rightarrow$ 4
        \item À 65 ns (front montant) : \texttt{count} $\rightarrow$ 5
        \item À 75 ns (front montant) : \texttt{count} $\rightarrow$ 6
        \item À 85 ns (front montant) : \texttt{count} $\rightarrow$ 7. Le signal \texttt{fin} passe \textbf{immédiatement à '1'} (logique combinatoire).
        \item À 95 ns (front montant) : \texttt{count} $\rightarrow$ 0 (bouclage car $N=8$). Le signal \texttt{fin} retombe immédiatement à '0'.
    \end{itemize}
    \item \textbf{Au-delà de 100 ns :}
    \begin{itemize}
        \item À 105 ns (front montant) : \texttt{count} $\rightarrow$ 1. 
        \item À 120 ns, \texttt{en} repasse à '0' (selon le testbench), le compteur s'arrête sur sa valeur courante.
    \end{itemize}
\end{itemize}
